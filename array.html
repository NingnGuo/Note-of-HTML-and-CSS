数组常见方法

push()方法将数据追加到数组末尾


var myArray = [["John", 23], ["cat", 2]];
document.write(myArray.push(["dog",3]));

输出：


[["John", 23], ["cat", 2],["dog",3]]

pop()方法用于去掉数组中最后一个元素


var array=[1,4,6];
document.write(array.pop());

输出：


1, 4

shift()移除数组中第一个元素


var array=[1,4,6];
document.write(array.shift());

输出：


4, 6

map方法会迭代数组中的每一个元素，并根据回调函数来处理每一个元素，并返回新的数组，但不会改变原数组


var oldArray = [1,2,3,4,5];
var newArray = oldArray.map(function(val){
    return val+3;
    });

输出：


4,5,6,7,8

reduce方法，迭代数组，并将结果累积到一个值中

reduce(回调函数（参数1，可选参数2){}），参数2用来设置累加器的初始值，若位设置，则从数组第一项开始，若设置则从第二项开始。


var array=[1,2,3];
var s=array.reduce(function(p,c){
  return p+c;
  });

输出：

6//如上方法是将数组中所有值相加

filter方法:迭代数组，按条件过滤出符合条件的元素

filter(function(当前迭代的项){})，回调函数返回true的值会保留在数组中，返回false的项会被过滤出数组


var array=[1,2,3,7];
var myarray=array.filter(function(val){
    return val<7;
    });

输出：


1,2,3//返回数组中小于7的元素

sort()排序


var a=[3,9,8,5,2,1];
a.sort();

输出：


1,2,5,8,9,3

从大到小排列数组元素


var array = [1, 12, 21, 2];
array.sort(function(a,b){
  return b-a;
});

reverse()翻转数组


var a=[1,4,6];
document.write(a.reverse());

输出：


6,4,1

concat()将两个数组合并


var a=[1,2,3];
var b=[4,5,6];
document.write(a.concat(b));

输出：


1,2,3,4,5,6

split()分割字符串


var string = "Split me into an array";
var array = [];
array = string.split(" ");
document.write(array);

输出：


"Split","me","into","an","array"//使用 split 方法来把字符串 string 分割为数组 array。

join()将数组用指定连接符连接成字符串


var array=["You","me"];
document.write(array.join("and"));

输出：


You and me
