"=="运算符与"==="

在js中，使用"=="运算符对两个不同的数据类型进行比较，一种类型要先转化为另一种类型。如下例子

        
           1   ==  1    // true
           1   ==  2    // false
           1   == '1'   // true
          "3"  ==  3    // true

===运算符，会比较元素的数值和数据类型

        
           3===3；//true
           3==='3';//false

!=与！==

！=与==是相反的

        
           1 != 2      // true
           1 != "1"    // false
           1 != '1'    // false
           1 != true   // false
           0 != false  // false

！==运算符与===是相反的

        
          3 !== 3   // false
          3 !== '3' // true
          4 !== 3   // true

switch-case语句，无法指定所有情况时，用default;当输出结果相同，但输入值不同时，如下1,2可以合在一起 

        
           function myTest(val) {
              var answer = "";
              switch(val){
                case 1:
                case 2:
                  answer="alpha";
                  break;
                case 3:
                  answer="gamma";
                  break;
                 }
                 default:
                 answer="beta";
                  break;
                return answer;  
                }
              myTest(1);

所有的比较运算符返回结果都是true或false，if/ else的返回结果也是true或false，所以关于比较的，可以直接返回比较的结果，例子如下： 

        
           function isEqual(a,b){
           if(a===b){
           return true;
            }else{
                return false;
             }
            }
            //我们可以做如下修改
            function isEqual(a,b){
            return a===b;
          }

当代码执行到return语句时，函数返回这个结果就执行结束了，不会执行return之后的语句。例子如下：函数将输出"Hello"到控制台、返回“World”，但没有输出"byebye".

        
           function myFun(){
           console.log("Hello");
           return "World";
           console.log("byebye");
         }

关于JS的对象

对象适合用来存储结构化数据，通过属性来访问和修改数据使用.或[]读取属性

当被访问对象的属性名中如果含有空格时，要使用[]来访问属性，且属性名要使用""或''括起来

        
           var obj={
           "name":"Jenny",
           "age":18,
           "sex":"female"
           "date time":"1997-01-01"
           };
         var nameValue=obj.name;
         var dateTime=obj.["date time"];

         obj.age=21;//使用.操作符更新对象的属性

         obj.city="QingDao";//使用.操作符给对象增加属性

         delete obj.sex;//删除对象的属性

检查一个对象是否存在某属性.hasOwnProperty(属性名),如下：函数checkObj检查myObj是否有checkProp属性，如果属性存在，返回属性对应的值，如果不存在，返回 "Not Found"。

        
          var myObj = {
            gift: "pony",
            pet: "kitten",
             bed: "sleigh"
          }; 
          function checkObj(checkProp) {
            if(myObj.hasOwnProperty(checkProp)){
              return myObj[checkProp];
            }else{ return "Not Found";}
            return "修改这里";
          }
          checkObj("gift");

           var car = {
            "wheels":4,
            "engines":1,
            "seats":5
          };
          //构造函数
          var Car=function(){
          this.wheels = 4;

            this.seats = 1;
                  }; 
        
        
调用构造函数(使用new关键字)来创建构造函数的实例，这时的实例可以像js的对象一样使用

        
          var myCar=new Car();

          var Car=function(wheels,seats.engines){
              this.wheels=wheels;

              this.engines=engines;
        };

        //调用有参构造函数
